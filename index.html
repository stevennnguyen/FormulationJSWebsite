Setting Up

There are a variety of ways to get FormulationJs started on your project.

Cloning Git Repository
  1. Navigate to a directory of your choice.
  2. Open up a git cmd.
  3. git clone git@github.com:codazen/formulationjs.git
  4. npm install
  5. npm start
Downloading Bundle Kit

Simply go to our download and download the kit locally.

Checkbox


Code


class Checkbox extends React.Component {

  constructor(props) {
    super(props);
    this.state = {
      checked: this.props.checkbox.value
    };
    this._onChange = this._onChange.bind(this);
    this._labelClick = this._labelClick.bind(this);
  }

  _onChange() {
    this.setState({
      checked: !this.state.checked
    });
  }

  _labelClick() {
    this._onChange();
  }
  
  render() {
    var checkbox = this.props.checkbox;
    return (
      <div>
        <label htmlFor={checkbox.name} onClick={this._labelClick}>{checkbox.label}</label>
        <input type="checkbox" checked={this.state.checked} name={checkbox.name} onChange={this._onChange}/>
      </div>
    );
  }
}

export default Checkbox;

Dropdown List


Code


class Dropdown extends React.Component {

  constructor(props) {
    super(props);
  }

  render() {
    var options = this.props.options;
    return (
      <div>
        <select>
          {
            options.map(function (option, index) {
              return <option key={index} value={option.value}>{option.name}</option>;
            })
          }
        </select>
      </div>
    );
  }
}

export default Dropdown;
